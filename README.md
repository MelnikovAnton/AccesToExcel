##Настройка
Для настройки приложения необходимо создать файл 
app.properties в директории resources.
#### Пример
localDBpath = D:/IP.mdb

localExcelPath = D:/ip_plan.xlsx

remoteFileName = /RemoteDir/IP.mdb

ftp.server = serverName

ftp.port = 21

ftp.user = user 

ftp.pass = password  

## Инсталяция 
#### mvn install
